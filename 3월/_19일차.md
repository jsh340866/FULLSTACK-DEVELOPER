    *** 개별 form 내의 input 접근(name 부여) ***
    const joinForm = document.joinForm; : joinForm 찾기!
    joinForm.username.value = '홍길동'; : username이름의 input에 '홍길동'을 부여!
    joinForm.nickname.value = '티모';  : nickname이름의 input에 '티모'를 부여!

    *** 스크립트로 클래스 부여 및 이벤트 속성 추가! ***
    const btnEls = document.querySelectorAll('.btn');  : .btn클래스 모두 찾기!
    btnEls.forEach((btnEl,idx)=>{    : 
      console.log(idx,btnEl);
      btnEl.addEventListener('mouseover',(e)=>{btnEl.classList.add(`btn--style${idx+1}`)});
          : btn클래스에 btn--style$ 클래스 추가!
      btnEl.addEventListener('mouseleave',(e)=>{btnEl.classList.remove(`btn--style${idx+1}`)});
    });
